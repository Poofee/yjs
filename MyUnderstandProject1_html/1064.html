14     void ea1a(coor0,dof,nodvar,ubf,elem,matrix,f)
15     int dof,*nodvar;
16     double *ubf,*f;
17     struct coordinates coor0;
18     struct element elem;
19     struct matrice matrix;
20     {
21         /*变量解释：
22         dof：    该场每个节点的自由度数；
23         ntype：  该场单元类型数；
24         nnode:   某类型单元一个单元上的节点数+1(包括材料)；
25         kvar:    总自由度数；
26         nne：    某类型单元一个单元上的节点数；
27         dim：    坐标维数；
28         knode：  节点总数；
29         neq：    该场待求方程总数；
30         maxa：   该场总刚矩阵元素总数；
31         ubf：    边界值；
32         nmate：  某种单元类型对应的材料数；
33         nprmt：  某种类型单元材料对应的材料参数个数；
34         nelem：  该场某类型单元的单元总数；
35         mate[]： 材料参数值空间；
36         prmt[]:  前nprmt个元素存储单元的材料信息，后nne个元素存储单元的整体节点号；
37         coef[]:  耦合场信息；
38         node[]： 所有单元的单元拓扑关系空间；
39         lm[]:    单元中所有未知量方程号空间；
40         *nodvar：该场节点自由度对应的规格数或者方程号；
41         *numcol：总体刚度矩阵每行(或列)的起始元素在na中的位置；
42         *coor：  节点坐标空间；
43         *r:      单元节点坐标空间
44         *a：     总刚矩阵；
45         *na：    总刚去零元 一维变带宽存储矩阵；
46         *es：    单元刚度矩阵；
47         *em：    单元质量矩阵；
48         *ec：    单元阻尼矩阵；
49         *ef：    单元荷载向量。
50         */
51         int ntype,nnode,kvar;
52         int i,j,k,l,m,kk,ij,nn,mm,nr,nrw,ne,nne,numel,idof,jdof,
53             inod,jnod,nodi,nodj,inv,jnv;
54         int dim,knode,neq,*numcol,maxa,*na,node[500],lm[500];
55         double *a,mate[500],prmt[500],coef[500],*u;
56         double *coor,*r;
57         double *estifn;
58         double *es,*em,*ec,*ef;
59         dim   = coor0.dim;
60         knode = coor0.knode;
61         coor  = coor0.coor;
62         kvar  = knode*dof;
63         a = matrix.a;
64         na = matrix.na;
65         neq = matrix.neq;
66         maxa = matrix.maxa;
67         numcol = matrix.numcol;
68         for(i=1; i<=maxa; ++i)
69         {
70             a[i] = 0.0;
71         }
72         u = (double *) calloc(kvar,sizeof(double));
73         r = (double *) calloc(500,sizeof(double));
74         for(i=0; i<kvar; ++i)
75         {
76             u[i]=ubf[i];
77         }
78         numel = 0;
79         nn = 0;
80         mm = 0;
81         ntype = elem.ntype;
82         /*  计算总体刚度矩阵以及边界处理  */
83         for (ityp=1; ityp<=ntype; ++ityp)
84         {
85             nmate = elem.nmate[ityp];
86             nprmt = elem.nprmt[ityp];
87             /*  得到某类型单元的某材料的材料参数值，存储在mate中  */
88             for (i=1; i<=nmate; ++i)
89                 for(j=1; j<=nprmt; ++j)
90                     mate[(i-1)*nprmt+j] = elem.mate[++mm];
91             nelem = elem.nelem[ityp];
92             nnode = elem.nnode[ityp];
93             nne   = nnode-1;
94             for(ne=1; ne<=nelem; ++ne)
95             {
96                 /*  读取单元节点拓扑关系，存储在node中  */
97                 for (j=1; j<=nnode; ++j)
98                     node[j] = elem.node[++nn];
99                 /*  通过每个类型单元的第一个单元的自由度规格数确定该类型单元的单刚，单质等矩阵大小  */
100                 if  (ne==1)
101                 {
102                     k=0;
103                     for(j=1; j<=nne; ++j)
104                     {
105                         jnod = node[j];
106                         if(jnod>0)
107                             for(l=1; l<=dof; ++l)
108                                 if(nodvar[(l-1)*(knode)+jnod-1] !=0 )
109                                     k++;
110                     }
111                     kk = k*k;
112                     es = (double *) calloc(kk,sizeof(double));
113                     em = (double *) calloc(k+1,sizeof(double));
114                     ec = (double *) calloc(k+1,sizeof(double));
115                     ef = (double *) calloc(k+1,sizeof(double));
116                     estifn = (double *) calloc(kk+1,sizeof(double));
117                 }
118                 /*  读取单元信息：单元的整体节点号(存储在prmt中)，单元节点坐标(存储在r中)  */
119                 for (j=1; j<=nne; ++j)
120                 {
121                     jnod=node[j];
122                     if(jnod<0)
123                         jnod = -jnod;
124                     prmt[nprmt+j] = jnod;
125                     for(i=1; i<=dim; ++i)
126                         r[(i-1)*(nne)+j-1] = coor[(i-1)*(knode)+jnod-1];
127                 }
128                 /*  读取单元信息：单元的材料信息(存储在prmt中)  */
129                 imate = node[nnode];
130                 for (j=1; j<=nprmt; ++j)
131                     prmt[j] = mate[(imate-1)*nprmt+j];
132                 /*  调用单元子程序，计算单元刚度矩阵,单元质量矩阵,单元阻尼矩阵和单元载荷向量  */
133                 switch (ityp)
134                 {
135                 case 1 :
136                     aeq4g2(r,coef,prmt,es,em,ec,ef,ne);
137                     break;
138                 case 2 :
139                     aet3g2(r,coef,prmt,es,em,ec,ef,ne);
140                     break;
141                 case 3 :
142                     aeq9g3(r,coef,prmt,es,em,ec,ef,ne);
143                     break;
144                 case 4 :
145                     aet6(r,coef,prmt,es,em,ec,ef,ne);
146                     break;
147                 }
148                 /*  组合单元刚度矩阵、单元质量矩阵、单元阻尼矩阵得到单元矩阵，对应sch算法文件中的matrix行  */
149                 for (i=1; i<=k; ++i)
150                     for (j=1; j<=k; ++j)
151                         estifn[(i-1)*(k)+j-1]=0.0;
152                 for (i=1; i<=k; ++i)
153                 {
154                     for (j=1; j<=k; ++j)
155                     {
156                         estifn[(i-1)*(k)+j-1]=estifn[(i-1)*(k)+j-1]
157                                               +es[(i-1)*(k)+j-1];
158                     }
159                     estifn[(i-1)*(k)+i-1]=estifn[(i-1)*(k)+i-1]
160                                           ;
161                 }
162                 i=0;
163                 /*  处理边界条件，得到单元矩阵右端项  */
164                 for (inod=1; inod<=nne; ++inod)
165                 {
166                     nodi = node[inod];
167                     for (idof=1; idof<=dof; ++idof)
168                     {
169                         inv = nodvar[(idof-1)*(knode)+nodi-1];
170                         if (inv==0)
171                             goto l600;
172                         /*  i中记录inv(规格数)不等于0的自由度数目  */
173                         i++;
174                         /*  lm中记录inv(规格数)不等于0的每个节点的每个自由度对应的方程号  */
175                         lm[i] = inv;
176                         /*  对于自由度规格数大于零的方程，进行荷载处理，对应sch算法文件中的force项  */
177                         if  (inv>0)
178                         {
179                             u[(idof-1)*(knode)+nodi-1] = u[(idof-1)*(knode)+nodi-1]
180                                                          +ef[i];
181                         }
182                         /*  处理强制边界条件  */
183                         j = 0;
184                         for (jnod=1; jnod<=nne; ++jnod)
185                         {
186                             nodj = node[jnod];
187                             for(jdof=1; jdof<=dof; ++jdof)
188                             {
189                                 jnv = nodvar[(jdof-1)*(knode)+nodj-1];
190                                 if (jnv==0)
191                                     goto l400;
192                                 j++;
193                                 if(jnv>0)
194                                 {
195                                     u[(jdof-1)*(knode)+nodj-1] = u[(jdof-1)*(knode)+nodj-1]
196                                                                  ;
197                                     /*  处理强制边界条件，对于自由度未知的那些行，其右端项要减去该行方程与已知自由度相关的项  */
198                                     if  (inv<0)
199                                         u[(jdof-1)*(knode)+nodj-1]=u[(jdof-1)*(knode)+nodj-1]-estifn[(i-1)*(k)+j-1]*u[(idof-1)*(knode)+nodi-1];
200                                 }
201     l400:
202                                 continue;
203                             }
204                         }
205     l600:
206                         continue;
207                     }
208                 }
209                 /*  集成总体刚度矩阵  */
210                 adda(na,a,numcol,k,lm,estifn,neq,maxa);
211     l1000:
212                 continue;
213                 if(ne==nelem)
214                 {
215                     free(es);
216                     free(em);
217                     free(ec);
218                     free(ef);
219                     free(estifn);
220                 }
221             }
222             numel += nelem;
223         }
224         free(r);
225         /*  根据单元循环中得到的u，得到总体方程右端项并改为一维存储(存储在数组f中)  */
226         for (ij=1; ij<=neq; ++ij)
227         {
228             f[ij] = 0.0;
229         }
230         for (i=1; i<=dof; ++i)
231         {
232             for (j=1; j<=knode; ++j)
233             {
234                 ij = nodvar[(i-1)*(knode)+j-1];
235                 if  (ij>0)
236                     f[ij] += u[(i-1)*(knode)+j-1];
237             }
238         }
239         free(u);
240         return;
