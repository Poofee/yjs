14     int dof,*nodvar;
15     double **f;
16     struct coordinates coor0;
17     struct element elem;
18     struct matrice *matrix;

20         /*
21         变量解释：
22         nelem：  该场某类型单元的单元总数；
23         nnode：  某类型单元一个单元上的节点数+1(包括材料)。
24         neq：    该场待求方程总数；
25         maxcol： 总体刚度矩阵的最大列号；
26         maxa:    总刚非零元总数；
27         kvar：   总自由度数；
28         dim：    坐标维度；
29         knode：  节点总数；
30         dof：    该场每个节点的自由度数；
31         ntype：  该场单元类型数；
32         *coor：  节点坐标空间；
33         *nodvar：该场节点自由度对应的规格数或者方程号；
34         *numcol：总刚矩阵每行(或列)的起始元素在na中的位置；
35         *na：    总刚去零元,一维变带宽存储矩阵；
36         */
37         int nelem,nnode,neq;
38         int *numcol,*na,*mht;
39         int i,j,l,n,nn,ne,nne,inod,nodi,numel,idof,maxcol,ityp,inv,maxa,kvar;
40         int dim,knode,lm[500],node[500];
41         double *coor;
42         dim   = coor0.dim;
43         knode = coor0.knode;
44         coor  = coor0.coor;
45         kvar  = knode*dof;
46         neq=0;
47         /*  遍历所有节点上的所有自由度，得到每个未知量对应的方程号以及方程组的方程个数neq  */
48         for (j=1; j<=knode; ++j)
49             for (i=1; i<=dof; ++i)
50                 if (nodvar[(i-1)*(knode)+j-1] == 1)
51                     nodvar[(i-1)*(knode)+j-1] = ++neq;
52         /*  从节点未知量的方程号上处理节点的主从约束关系  */
53         for (j=1; j<=knode; ++j)
54             for (i=1; i<=dof; ++i)
55                 if (nodvar[(i-1)*(knode)+j-1] < -1)
56                 {
57                     n = -nodvar[(i-1)*(knode)+j-1]-1;
58                     nodvar[(i-1)*(knode)+j-1] = nodvar[(i-1)*(knode)+n-1];
59                 }
60         if (neq>0)
61         {
62             maxcol=maxt/neq;
63         }
64         else
65         {
66             printf("warning! there isn't any unknown variable on nodes!%c",endl);
67             maxcol=1;
68         }
69         na = (int *) calloc(maxt,sizeof(int));
70         (*matrix).jdiag=(int *) calloc(neq+1,sizeof(int));
71         mht = na;
72         numcol = (int *) calloc(neq+1,sizeof(int));
73         for (i=0; i<=neq; ++i)
74             numcol[i] = 0;
75         numel = 0;
76         nn = 0;
77         /*  遍历每种类型的所有单元，将单元刚度矩阵的每个元素（非零元）与总刚度矩阵的元素（非零元）一一对应  */
78         for (ityp=1; ityp<=elem.ntype; ++ityp)
79         {
80             nelem = elem.nelem[ityp];
81             nnode = elem.nnode[ityp];
82             for (i=1; i<=nelem; ++i)
83             {
84                 for (j=1; j<=nnode; ++j)
85                     node[j] = elem.node[++nn];
86                 nne = nnode-1;
87                 {
88                     l = 0;
89                     /*  在一个单元中，l表示该单元中待求方程数，并将第i方程对应的整体方程号赋给lm[i]  */
90                     for (inod=1; inod<=nne; ++inod)
91                     {
92                         nodi = node[inod];
93                         for (idof=1; idof<=dof; ++idof)
94                         {
95                             inv = nodvar[(idof-1)*(knode)+nodi-1];
96                             if (inv>0)
97                             {
98                                 l++;
99                                 lm[l] = inv;
100                             }
101                         }
102                     }
103                     numel++;
104                     /*  aclh函数：得到总刚矩阵非零元的位置(存储在*mht矩阵中)；
105                                  得到总刚矩阵每行非零元的个数(存储在*numcol中)  */
106                     if (l>0)
107                         aclh(neq,numcol,mht,l,lm,maxcol);
108                 }
109             }
110         }
111         /*  bclh函数：对mht矩阵按行排序，将其中的非零元素存储在na中；
112                       得到总体刚度矩阵每行的起始元素在na中的位置(存储在numcol中)  */
113         bclh(neq,numcol,mht,na,(*matrix).jdiag,lm,maxcol);
114         /*  把得到的总体刚度矩阵信息存储到matrix中去  */
115         maxa=numcol[neq];
116         (*matrix).maxa = maxa;
117         (*matrix).neq = neq;
118         (*matrix).na = (int *)   calloc(maxa+1,sizeof(int));
119         (*matrix).a = (double *) calloc(maxa+1,sizeof(double));
120         (*matrix).numcol = (int *)  calloc(neq+1,sizeof(int));
121         (*f) = (double *) calloc(neq+1,sizeof(double));
122         for (n=0; n<=neq; ++n) (*matrix).numcol[n] = numcol[n];
123         for (n=0; n<maxa; ++n) (*matrix).na[n+1]   = na[n];
124         free(numcol);
125         free(na);
126         return;
127     }
