347     static double ftran_coor(double *refc,int n)
348     /* coordinate transfer function caculation */
349     {
350         double rx,ry,fval;
351         double x[7],y[7];
352         int j;
353         for (j=1; j<=6; ++j)
354         {
355             x[j]=coorr[(1-1)*(6)+j-1];
356             y[j]=coorr[(2-1)*(6)+j-1];
357         }
358         rx=refc[1];
359         ry=refc[2];
360         switch (n)
361         {
362         case 1:
363             fval=
364                 +(+rx*(+2.*rx-1.))*x[1]
365                 +(+rx*ry*4.)*x[4]
366                 +(+ry*(+2.*ry-1.))*x[2]
367                 +(+ry*(+1.-rx-ry)*4.)*x[5]
368                 +(+(+1.-2.*rx-2.*ry)*(+1.-rx-ry))*x[3]
369                 +(+rx*(+1.-rx-ry)*4.)*x[6];
370             break;
371         case 2:
372             fval=
373                 +(+rx*(+2.*rx-1.))*y[1]
374                 +(+rx*ry*4.)*y[4]
375                 +(+ry*(+2.*ry-1.))*y[2]
376                 +(+ry*(+1.-rx-ry)*4.)*y[5]
377                 +(+(+1.-2.*rx-2.*ry)*(+1.-rx-ry))*y[3]
378                 +(+rx*(+1.-rx-ry)*4.)*y[6];
379             break;
380             //default:
381         }
382         return fval;
383     }
